#### 1.2 实现权限中间件

```go
package middleware

import (
    "github.com/gin-gonic/gin"
)

// RequireAuth 认证中间件
func RequireAuth() gin.HandlerFunc {
    return func(c *gin.Context) {
        token := c.GetHeader("Authorization")
        if token == "" {
            c.JSON(401, gin.H{"error": "未登录"})
            c.Abort()
            return
        }
        
        // 验证 JWT
        claims, err := validateToken(token)
        if err != nil {
            c.JSON(401, gin.H{"error": "token 无效"})
            c.Abort()
            return
        }
        
        // 存储用户信息
        c.Set("user_id", claims.UserID)
        c.Set("username", claims.Username)
        c.Set("role", claims.Role)
        
        c.Next()
    }
}

// RequirePermission 权限检查中间件
func RequirePermission(permission string) gin.HandlerFunc {
    return func(c *gin.Context) {
        role := c.GetString("role")
        
        if !hasPermission(role, permission) {
            c.JSON(403, gin.H{"error": "权限不足"})
            c.Abort()
            return
        }
        
        c.Next()
    }
}

func hasPermission(role string, permission string) bool {
    permissions, ok := RolePermissions[role]
    if !ok {
        return false
    }
    
    for _, p := range permissions {
        if p == permission {
            return true
        }
    }
    return false
}

// 使用示例
func SetupRoutes(r *gin.Engine) {
    api := r.Group("/api/v1")
    api.Use(RequireAuth())
    
    // 提交审查（需要 developer 权限）
    api.POST("/reviews", 
        RequirePermission(PermSubmitReview),
        controller.SubmitReview)
    
    // 查看报告（需要 view 权限）
    api.GET("/reviews/:id",
        RequirePermission(PermViewReport),
        controller.GetReviewReport)
    
    // 删除报告（需要 admin 权限）
    api.DELETE("/reviews/:id",
        RequirePermission(PermDeleteReport),
        controller.DeleteReview)
}
```

func (c *ReviewController) GetReviewReport(ctx *gin.Context) {
    reviewID := ctx.Param("id")

        }
    }

package middleware

// AuditLog 审计日志中间件
func AuditLog(action string) gin.HandlerFunc {
    return func(c *gin.Context) {
        startTime := time.Now()
        
        // 记录请求信息
        userID := c.GetInt("user_id")
        username := c.GetString("username")
        resource := c.Request.URL.Path
        ip := c.ClientIP()
        userAgent := c.Request.UserAgent()
        
        // 执行请求
        c.Next()
        
        // 记录审计日志
        status := "success"
        errorMsg := ""
        
        if len(c.Errors) > 0 {
            status = "failed"
            errorMsg = c.Errors.String()
        } else if c.Writer.Status() >= 400 {
            status = "failed"
        }
        
        log := &AuditLog{
            UserID:    userID,
            Username:  username,
            Action:    action,
            Resource:  resource,
            IP:        ip,
            UserAgent: userAgent,
            Status:    status,
            Error:     errorMsg,
            CreatedAt: time.Now(),
            Duration:  time.Since(startTime),
        }
        
        // 异步保存审计日志
        go saveAuditLog(log)
    }
}

// 使用示例
api.POST("/reviews",
    RequireAuth(),
    RequirePermission(PermSubmitReview),

### 3. Token 成本记录（星期二）

type TokenUsage struct {
    ID            int       `json:"id"`
    UserID        int       `json:"user_id"`
    ReviewID      string    `json:"review_id"`
    Model         string    `json:"model"`         // gpt-4o, gpt-4o-mini
    PromptTokens  int       `json:"prompt_tokens"`
    OutputTokens  int       `json:"completion_tokens"`
    TotalTokens   int       `json:"total_tokens"`
    Cost          float64   `json:"cost"`          // 美元
    CreatedAt     time.Time `json:"created_at"`
}

// 记录 Token 使用
func recordTokenUsage(ctx context.Context, userID int, reviewID string, usage openai.Usage, model string) error {
    cost := calculateCost(usage, model)
    
    record := &TokenUsage{
        UserID:       userID,
        ReviewID:     reviewID,
        Model:        model,
        PromptTokens: usage.PromptTokens,
        OutputTokens: usage.CompletionTokens,
        TotalTokens:  usage.TotalTokens,
        Cost:         cost,
        CreatedAt:    time.Now(),
    }
    
    return db.Create(record).Error
}

// 计算成本（2024 年价格）
func calculateCost(usage openai.Usage, model string) float64 {
    var inputCost, outputCost float64
    
    switch model {
    case "gpt-4o":
        inputCost = 5.0 / 1_000_000   // $5 / 1M tokens
        outputCost = 15.0 / 1_000_000 // $15 / 1M tokens
    case "gpt-4o-mini":
        inputCost = 0.15 / 1_000_000  // $0.15 / 1M tokens
        outputCost = 0.6 / 1_000_000  // $0.6 / 1M tokens
    }
    
    return float64(usage.PromptTokens)*inputCost + 
           float64(usage.CompletionTokens)*outputCost
}

// 查询用户成本
func GetUserCost(ctx context.Context, userID int, startDate, endDate time.Time) (float64, error) {
    var totalCost float64
    
    err := db.Model(&TokenUsage{}).
        Where("user_id = ? AND created_at BETWEEN ? AND ?", userID, startDate, endDate).
        Select("SUM(cost)").
        Scan(&totalCost).Error
    
    return totalCost, err
}

### 4. 限流与降级（星期三）

#### 4.1 限流中间件

```go
package middleware

import (
    "golang.org/x/time/rate"
    "sync"
)

// RateLimiter 限流器
type RateLimiter struct {
    limiters map[int]*rate.Limiter  // 用户ID -> 限流器
    mu       sync.RWMutex
    rate     rate.Limit
    burst    int
}

func NewRateLimiter(r rate.Limit, b int) *RateLimiter {
    return &RateLimiter{
        limiters: make(map[int]*rate.Limiter),
        rate:     r,
        burst:    b,
    }
}

func (rl *RateLimiter) getLimiter(userID int) *rate.Limiter {
    rl.mu.Lock()
    defer rl.mu.Unlock()
    
    limiter, exists := rl.limiters[userID]
    if !exists {
        limiter = rate.NewLimiter(rl.rate, rl.burst)
        rl.limiters[userID] = limiter
    }
    
    return limiter
}

// RateLimit 限流中间件
func RateLimit(limiter *RateLimiter) gin.HandlerFunc {
    return func(c *gin.Context) {
        userID := c.GetInt("user_id")
        
        if !limiter.getLimiter(userID).Allow() {
            c.JSON(429, gin.H{
                "error": "请求过于频繁，请稍后再试",
            })
            c.Abort()
            return
        }
        
        c.Next()
    }
}

// 使用示例
func main() {
    r := gin.Default()
    
    // 每个用户每秒最多 5 个请求，突发 10 个
    limiter := NewRateLimiter(5, 10)
    
    api := r.Group("/api/v1")
    api.Use(RequireAuth())
    api.Use(RateLimit(limiter))
    
    api.POST("/reviews", controller.SubmitReview)
}
```

// 降级配置
type DegradationConfig struct {
    EnableAI       bool    `json:"enable_ai"`        // 是否启用 AI 分析
    MaxConcurrency int     `json:"max_concurrency"`  // 最大并发数
    Timeout        int     `json:"timeout"`          // 超时时间（秒）
    QuickMode      bool    `json:"quick_mode"`       // 快速模式（只静态分析）
}

var degradationConfig = &DegradationConfig{
    EnableAI:       true,
    MaxConcurrency: 10,
    Timeout:        300,
    QuickMode:      false,
}

// Review 审查代码（带降级）
func (a *CodeReviewAgent) Review(ctx context.Context, req ReviewRequest) (*ReviewReport, error) {
    report := &ReviewReport{
        ProjectName: req.ProjectName,
        CommitHash:  req.CommitHash,
    }
    
    // 1. 静态分析（必须执行）
    lintIssues, _ := a.staticAnalyzer.AnalyzeWithGolangCI(req.ProjectPath)
    report.LintIssues = lintIssues
    
    secIssues, _ := a.staticAnalyzer.AnalyzeWithGosec(req.ProjectPath)
    report.SecurityIssues = secIssues
    
    // 2. AI 分析（可降级）
    if degradationConfig.EnableAI && !degradationConfig.QuickMode {
        // 带超时的 AI 分析
        ctx, cancel := context.WithTimeout(ctx, time.Duration(degradationConfig.Timeout)*time.Second)
        defer cancel()
        
        aiIssues, err := a.aiAnalyzer.AnalyzeSecurity(ctx, code, file)
        if err != nil {
            // AI 分析失败，记录但不报错
            log.Printf("AI 分析失败，已降级: %v", err)
        } else {
            report.AISecurityIssues = aiIssues
        }
    }
    
    return report, nil
}
