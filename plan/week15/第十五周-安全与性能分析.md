### 1. Web 安全基础（星期一-星期二）

#### 1.1 SQL 注入（SQL Injection）

**漏洞原理**
```go
// ❌ 危险：直接拼接 SQL
func GetUser(username string) (*User, error) {
    query := "SELECT * FROM users WHERE username = '" + username + "'"
    // 如果 username = "admin' OR '1'='1"
    // SQL 变成: SELECT * FROM users WHERE username = 'admin' OR '1'='1'
    return db.Query(query)
}
```

**防御方法**
```go
// ✅ 使用参数化查询
func GetUser(username string) (*User, error) {
    query := "SELECT * FROM users WHERE username = ?"
    return db.Query(query, username)  // GORM 会自动转义
}

// ✅ GORM 正确使用
func GetUserByName(username string) (*User, error) {
    var user User
    // GORM 自动使用预编译语句
    err := db.Where("username = ?", username).First(&user).Error
    return &user, err
}

// ❌ GORM 错误使用
func GetUserDangerous(username string) (*User, error) {
    var user User
    // 危险！直接拼接
    err := db.Where("username = '" + username + "'").First(&user).Error
    return &user, err
}
```

- [ ] 尝试用恶意输入测试接口

#### 1.2 XSS（跨站脚本攻击）

**漏洞原理**
```go
// 用户输入: <script>alert('XSS')</script>
// 如果直接返回到前端并渲染，脚本会执行
```

**防御方法**
```go
import "html"

// ✅ 对用户输入进行 HTML 转义
func CreateComment(content string) error {
    escaped := html.EscapeString(content)
    // <script> 变成 &lt;script&gt;
    return db.Create(&Comment{Content: escaped}).Error
}

// ✅ 在响应中设置安全的 Content-Type
func GetComment(c *gin.Context) {
    c.Header("Content-Type", "application/json; charset=utf-8")
    c.Header("X-Content-Type-Options", "nosniff")
    c.JSON(200, comment)
}
```

**CSP（内容安全策略）**
```go
// 设置 CSP 头
func CSPMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        c.Header("Content-Security-Policy", 
            "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline'")
        c.Next()
    }
}
```

#### 1.3 CSRF（跨站请求伪造）

**漏洞原理**
- 攻击者诱导用户点击恶意链接
- 利用用户已登录的 Cookie 发起请求

**防御方法**
```go
// ✅ 使用 CSRF Token
import "github.com/utrack/gin-csrf"

func main() {
    r := gin.Default()
    
    // CSRF 中间件
    r.Use(csrf.Middleware(csrf.Options{
        Secret: "your-secret-key",
        ErrorFunc: func(c *gin.Context) {
            c.JSON(403, gin.H{"error": "CSRF token invalid"})
            c.Abort()
        },
    }))
    
    r.POST("/transfer", TransferMoney)
}

// ✅ 检查 Referer 头
func RefererCheck() gin.HandlerFunc {
    return func(c *gin.Context) {
        referer := c.GetHeader("Referer")
        if !strings.HasPrefix(referer, "https://yourdomain.com") {
            c.JSON(403, gin.H{"error": "Invalid referer"})
            c.Abort()
            return
        }
        c.Next()
    }
}

// ✅ SameSite Cookie
func SetCookie(c *gin.Context, token string) {
    c.SetSameSite(http.SameSiteStrictMode)
    c.SetCookie("token", token, 3600, "/", "", true, true)
}
```

#### 1.4 越权访问

**水平越权**
```go
// ❌ 危险：只检查登录，不检查权限
func GetOrder(c *gin.Context) {
    orderID := c.Param("id")
    order, _ := orderRepo.GetByID(orderID)
    c.JSON(200, order)  // 任何登录用户都能查看任何订单
}

// ✅ 检查订单归属
func GetOrder(c *gin.Context) {
    orderID := c.Param("id")
    userID := c.GetInt("user_id")  // 从 JWT 获取
    
    order, err := orderRepo.GetByID(orderID)
    if err != nil {
        c.JSON(404, gin.H{"error": "Order not found"})
        return
    }
    
    // 检查订单是否属于当前用户
    if order.UserID != userID {
        c.JSON(403, gin.H{"error": "Permission denied"})
        return
    }
    
    c.JSON(200, order)
}
```

**垂直越权**
```go
// ❌ 危险：普通用户能调用管理员接口
func DeleteUser(c *gin.Context) {
    userID := c.Param("id")
    userRepo.Delete(userID)
}

// ✅ 角色权限检查
func AdminOnly() gin.HandlerFunc {
    return func(c *gin.Context) {
        role := c.GetString("role")
        if role != "admin" {
            c.JSON(403, gin.H{"error": "Admin only"})
            c.Abort()
            return
        }
        c.Next()
    }
}

// 使用中间件保护
r.DELETE("/users/:id", AdminOnly(), DeleteUser)
```

#### 1.5 敏感信息泄露

**常见问题**
```go
// ❌ 返回密码字段
type User struct {
    ID       int    `json:"id"`
    Username string `json:"username"`
    Password string `json:"password"`  // 危险！
}

// ✅ 使用 DTO 或 json tag
type User struct {
    ID       int    `json:"id"`
    Username string `json:"username"`
    Password string `json:"-"`  // 不序列化
}

type UserDTO struct {
    ID       int    `json:"id"`
    Username string `json:"username"`
    // 不包含敏感字段
}

// ❌ 详细的错误信息
c.JSON(500, gin.H{"error": err.Error()})  // 可能泄露数据库结构

// ✅ 通用错误信息
c.JSON(500, gin.H{"error": "Internal server error"})
log.Printf("Error details: %v", err)  // 只记录到日志
```

**安全日志**
```go
// ❌ 记录敏感信息
log.Printf("User login: username=%s, password=%s", username, password)

// ✅ 脱敏日志
log.Printf("User login: username=%s", username)
log.Printf("Payment: card=****%s", card[len(card)-4:])
```

#### 2.2 CPU 性能分析

**示例：发现热点函数**
```go
// 模拟 CPU 密集型操作
func SlowFunction() {
    for i := 0; i < 1000000; i++ {
        _ = fmt.Sprintf("number: %d", i)
    }
}

// 优化后
var pool = sync.Pool{
    New: func() interface{} {
        return new(bytes.Buffer)
    },
}

func FastFunction() {
    buf := pool.Get().(*bytes.Buffer)
    defer func() {
        buf.Reset()
        pool.Put(buf)
    }()
    
    for i := 0; i < 1000000; i++ {
        buf.WriteString("number: ")
        buf.WriteString(strconv.Itoa(i))
    }
}
```

#### 2.3 内存分析

**查找内存泄露**
```go
// ❌ 内存泄露：goroutine 泄露
func LeakExample() {
    ch := make(chan int)
    go func() {
        val := <-ch  // 永远阻塞，goroutine 泄露
        fmt.Println(val)
    }()
}

// ❌ 内存泄露：全局 map 不清理
var cache = make(map[string][]byte)

func CacheData(key string, data []byte) {
    cache[key] = data  // 永不删除，内存持续增长
}
```

**内存优化**
```go
// 使用对象池
var bufferPool = sync.Pool{
    New: func() interface{} {
        return new(bytes.Buffer)
    },
}

func ProcessData(data []byte) []byte {
    buf := bufferPool.Get().(*bytes.Buffer)
    defer bufferPool.Put(buf)
    
    buf.Reset()
    buf.Write(data)
    // process...
    return buf.Bytes()
}

// 及时释放大对象
func ProcessLargeFile(filename string) error {
    data, err := os.ReadFile(filename)
    if err != nil {
        return err
    }
    
    // 处理数据
    process(data)
    
    data = nil  // 帮助 GC
    runtime.GC()  // 建议 GC（不是强制）
    
    return nil
}
```

#### 2.4 慢 SQL 排查

**启用 GORM 日志**
```go
db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{
    Logger: logger.Default.LogMode(logger.Info),  // 记录所有 SQL
})

// 或只记录慢查询
db.Logger = logger.Default.LogMode(logger.Warn).SlowThreshold(200 * time.Millisecond)
```

**常见慢 SQL 问题**
```go
// ❌ N+1 问题
func GetOrdersWithItems() ([]Order, error) {
    var orders []Order
    db.Find(&orders)
    
    for i := range orders {
        db.Model(&orders[i]).Association("Items").Find(&orders[i].Items)
        // 每个订单查询一次，N 次查询
    }
    return orders, nil
}

// ✅ 使用 Preload
func GetOrdersWithItems() ([]Order, error) {
    var orders []Order
    err := db.Preload("Items").Find(&orders).Error
    // 只有 2 次查询：订单 + 所有商品
    return orders, err
}

// ❌ 缺少索引
db.Where("email = ?", email).First(&user)
// 如果 email 字段没有索引，会全表扫描

// ✅ 添加索引
type User struct {
    ID    int    `gorm:"primaryKey"`
    Email string `gorm:"index"`  // 添加索引
}

// ❌ SELECT *
db.Find(&users)  // 查询所有字段

// ✅ 只查询需要的字段
db.Select("id, username, email").Find(&users)
```

**分析 SQL 执行计划**
```sql
-- 在 MySQL 中查看执行计划
EXPLAIN SELECT * FROM users WHERE email = 'test@example.com';

-- 查看索引使用情况
SHOW INDEX FROM users;
```

### 3. 性能优化实战（星期四-星期五）

#### 3.1 接口性能优化

**优化前**
```go
func GetProductList(c *gin.Context) {
    var products []Product
    
    // 慢查询：没用索引、查询所有字段
    db.Order("created_at DESC").Limit(20).Find(&products)
    
    // N+1 问题：每个商品查询分类
    for i := range products {
        db.First(&products[i].Category, products[i].CategoryID)
    }
    
    c.JSON(200, products)
}
```

**优化后**
```go
func GetProductList(c *gin.Context) {
    cacheKey := "products:list:page:1"
    
    // 1. 查缓存
    if cached, err := redisClient.Get(ctx, cacheKey); err == nil {
        var products []Product
        json.Unmarshal([]byte(cached), &products)
        c.JSON(200, products)
        return
    }
    
    // 2. 优化查询
    var products []Product
    err := db.
        Select("id, name, price, category_id").  // 只查需要的字段
        Preload("Category").                      // 解决 N+1
        Where("status = ?", "active").            // 添加索引
        Order("created_at DESC").
        Limit(20).
        Find(&products).Error
    
    if err != nil {
        c.JSON(500, gin.H{"error": "Failed to fetch products"})
        return
    }
    
    // 3. 写缓存
    data, _ := json.Marshal(products)
    redisClient.Set(ctx, cacheKey, data, 5*time.Minute)
    
    c.JSON(200, products)
}
```

**对比结果**
```
优化前：800ms (数据库查询 700ms + 序列化 100ms)
优化后：50ms (缓存命中) / 150ms (缓存未命中)
```

#### 3.2 并发优化

**优化前：串行处理**
```go
func ProcessOrders(orderIDs []int) error {
    for _, id := range orderIDs {
        order, _ := orderRepo.GetByID(id)
        // 处理订单
        processOrder(order)  // 每个 500ms
    }
    // 100 个订单需要 50 秒
    return nil
}
```

**优化后：并发处理**
```go
func ProcessOrders(orderIDs []int) error {
    var wg sync.WaitGroup
    errCh := make(chan error, len(orderIDs))
    sem := make(chan struct{}, 10)  // 限制并发数
    
    for _, id := range orderIDs {
        wg.Add(1)
        go func(orderID int) {
            defer wg.Done()
            sem <- struct{}{}        // 获取信号量
            defer func() { <-sem }() // 释放信号量
            
            order, err := orderRepo.GetByID(orderID)
            if err != nil {
                errCh <- err
                return
            }
            processOrder(order)
        }(id)
    }
    
    wg.Wait()
    close(errCh)
    
    // 100 个订单只需要 5 秒（10 并发）
    return nil
}
```
